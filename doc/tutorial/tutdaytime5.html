Daytime.5 - A synchronous UDP daytime server

This tutorial program shows how to use asio to implement a server application with UDP.

#include <ctime>
#include <iostream>
#include <string>
#include "asio.hpp"

int main()
{
  try
  {
    asio::demuxer demuxer;

Create an asio::datagram_socket object to receive requests on UDP port 13.

    asio::datagram_socket socket(demuxer, asio::ipv4::udp::endpoint(13));

Wait for a client to initiate contact with us. The remote_endpoint object will be populated by asio::datagram_socket::receive_from().

    for (;;)
    {
      char recv_buf[1];
      asio::ipv4::udp::endpoint remote_endpoint;
      socket.receive_from(
          asio::buffers(recv_buf, sizeof(recv_buf)), 0, remote_endpoint,
          asio::throw_error_if(asio::the_error != asio::error::message_size));

Determine what we are going to send back to the client.

      using namespace std; // For time_t, time and ctime.
      time_t now = time(0);
      std::string msg = ctime(&now);

Send the response to the remote_endpoint.

      socket.send_to(asio::buffers(msg.c_str(), msg.length()), 0,
          remote_endpoint, asio::ignore_error());
    }
  }

Finally, handle any exceptions.

  catch (asio::error& e)
  {
    std::cerr << e << std::endl;
  }

  return 0;
}
