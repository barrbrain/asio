buffered_read_stream::peek

Peek at the incoming data on the stream. Returns the number of bytes read. Throws an exception on failure. 

template<
    typename MutableBufferSequence>
std::size_t peek(
    const MutableBufferSequence & buffers);

template<
    typename MutableBufferSequence>
std::size_t peek(
    const MutableBufferSequence & buffers,
    asio::error_code & ec);
