async_read_until

Start an asynchronous operation to read data into a streambuf until a delimiter is encountered. 

template<
    typename AsyncReadStream,
    typename Allocator,
    typename ReadHandler>
void async_read_until(
    AsyncReadStream & s,
    asio::basic_streambuf< Allocator > & b,
    char delim,
    ReadHandler handler);
